🧭 app/routes.py (Main Logic)

from flask import render_template, request, jsonify from .model import predict_score from . import app @app.route('/', methods=['GET', 'POST']) def index(): if request.method == 'POST': try: hours = float(request.form['hours']) result = predict_score(hours) return render_template('result.html', hours=hours, result=result) except Exception as e: return render_template('error.html', error=str(e)) return render_template('index.html') @app.route('/api/predict', methods=['POST']) def api_predict(): data = request.get_json() hours = data.get('hours') if hours is None: return jsonify({'error': 'Missing hours'}), 400 pred = predict_score(float(hours)) return jsonify({'prediction': pred, 'hours': hours})

🧠 app/model.py (Load Model)

import joblib import os model_path = os.path.join(os.path.dirname(__file__), '..', 'models', 'student_model.pkl') model = joblib.load(model_path) def predict_score(hours): # Simple linear regression: score = 10 + 5*hours if hours < 0 or hours > 24: raise ValueError("Hours must be 0-24") return round(model.predict([[hours]])[0], 2)

📓 notebooks/train_model.ipynb

# Train a simple linear regression model import numpy as np from sklearn.linear_model import LinearRegression import joblib # Generate synthetic data: score = 10 + 5*hours + noise np.random.seed(42) hours = np.random.uniform(1, 10, 200).reshape(-1, 1) scores = 10 + 5 * hours + np.random.normal(0, 2, hours.shape) model = LinearRegression() model.fit(hours, scores) # Save the model joblib.dump(model, '../models/student_model.pkl') # Expected: coef ≈ 5, intercept ≈ 10

⚡ How to Run

1. Create the folder structure above.

2. Copy each file content into the correct file.

3. Install dependencies:

pip install -r requirements.txt

4. Train the model (run the notebook or just use the code above):

python -c "import numpy as np; from sklearn.linear_model import LinearRegression; import joblib; np.random.seed(42); hours = np.random.uniform(1,10,200).reshape(-1,1); scores = 10 + 5*hours + np.random.normal(0,2,hours.shape); model = LinearRegression().fit(hours,scores); joblib.dump(model, 'models/student_model.pkl')"

5. Run the app:

python run.py

6. Open http://localhost:5000 in your browser.

7. Test the API:

curl -X POST -H "Content-Type: application/json" -d '{"hours":5.5}' http://localhost:5000/api/predict
